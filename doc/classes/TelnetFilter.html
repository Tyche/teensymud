Class TelnetFilter
In: lib/filter.rb
Parent: Filter

The TelnetFilter class implements the Telnet protocol.

This implements most of basic Telnet as per RFC 854/855/1143 and options in RFC 857/858/1073/1091

     # File lib/filter.rb, line 90
 90:   def filter_in(str)
 91:     buf = ""
 92:     return buf if str.nil? || str.empty?
 93:     @sc ? @sc.concat(str) : @sc = StringScanner.new(str)
 94:     while b = @sc.getbyte
 95:       case mode?
 96:       when :cr
 97:         # handle CRLF and CRNUL by swallowing what follows CR and
 98:         # insertion of LF
 99:         buf << LF.chr
100:         set_mode(:normal)
101:       when :cmd
102:         case b[0]
103:         when IAC
104:           # IAC escapes IAC
105:           buf << IAC.chr
106:           set_mode(:normal)
107:         when AYT
108:           @log.debug("(#{@conn.object_id}) AYT sent")
109:           @conn.sendmsg("TeensyMUD is here.\n")
110:           set_mode(:normal)
111:         when AO, IP, GA, NOP, BRK  # not implemented or ignored
112:           @log.debug("(#{@conn.object_id}) AO, IP, GA, NOP or BRK sent")
113:           set_mode(:normal)
114:         when EC
115:           @log.debug("(#{@conn.object_id}) EC sent")
116:           buf.slice!(-1)
117:           set_mode(:normal)
118:         when EL
119:           @log.debug("(#{@conn.object_id}) EL sent")
120:           p = buf.rindex("\n")
121:           p ? buf.slice!(pos+1..-1) : buf = ""
122:           set_mode(:normal)
123:         when DO, DONT, WILL, WONT
124:           opt = @sc.getbyte
125:           if opt.nil?
126:             @sc.peep
127:             break
128:           end
129:           case b[0]
130:           when WILL
131:             replies_him(opt[0],true)
132:           when WONT
133:             replies_him(opt[0],false)
134:           when DO
135:             requests_us(opt[0],true)
136:           when DONT
137:             requests_us(opt[0],false)
138:           end
139:           set_mode(:normal)
140:         when SB
141:           opt = @sc.getbyte
142:           if opt.nil? || @sc.check_until(/#{IAC.chr}#{SE.chr}/).nil?
143:             @sc.peep
144:             break
145:           end
146:           data = @sc.scan_until(/#{IAC.chr}#{SE.chr}/).chop.chop
147:           sub_neg(opt[0],data)
148:           set_mode(:normal)
149:         else
150:           @log.debug("(#{@conn.object_id}) Unknown Telnet command - #{b[0].to_s}")
151:           set_mode(:normal)
152:         end
153:       when :normal
154:         case b[0]
155:         when CR
156:           set_mode(:cr)
157:         when IAC
158:           set_mode(:cmd)
159:         when BS
160:           buf.slice!(-1)
161:         else
162:           buf << b
163:         end
164:       end
165:     end  # while b
166:     @sc = nil if @sc.eos?
167:     buf
168:   end
